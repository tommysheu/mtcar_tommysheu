Slide 2: MPG vs. Automatic/Manual Transmission

data(mtcars)
par(mfrow = c(1, 2))
plot(mpg ~ am, mtcars)
fit1 <- lm(mpg ~ am, data = mtcars)
abline(fit1, col = "red")
plot(residuals(fit1), type = "h", main = "model 1: mpg ~ am", xlab = "car index", 
    ylab = "residuals", )
